逆天舍友的链表代码

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// 定义链表节点结构
struct node
{
int data;
int count;
struct node* next;
};

// 创建新节点
struct node* create_node(int value, int count)
{
struct node* newNode = (struct node*)malloc(sizeof(struct node));
if (newNode == NULL)
{
printf(“内存分配失败\n”);
exit(1);
}
newNode->data = value;
newNode->count = count;
newNode->next = NULL;
return newNode;
}

// 打印链表
void print_list(struct node* head)
{
for (struct node* p = head; p != NULL; p = p->next)
{
printf(“%d “, p->data);
}
printf(“\n”);
}

// 在链表的任意位置后插入新节点
void insert_node(struct node* position, int value)
{
if (position == NULL)
{
printf(“前一个节点不能为空\n”);
return;
}
struct node* newNode = create_node(value, position->count + 1);
newNode->next = position->next;
position->next = newNode;

// 更新后续节点的计数
for (struct node* p = newNode->next; p != NULL; p = p->next) 
{
    p->count++;
}

}

// 删除节点
void delete_node(struct node** head, int value) {
struct node* p = head;
struct node q = NULL;
int positions[100];
int count = 0;
int index = 1;

// 找到所有匹配节点的位置
while (p != NULL) {
    if (p->data == value) {
        positions[count] = index;
        count++;
    }
    q = p;
    p = p->next;
    index++;
}

if (count == 0) {
    printf("未找到要删除的节点\n");
    return;
}

printf("找到%d个节点,位置分别为:", count);
for (int i = 0; i < count; i++) {
    printf("%d ", positions[i]);
}
printf("\n");

int del_pos;
printf("请输入要删除的节点位置(从1开始): ");
scanf("%d", &del_pos);

if (del_pos < 1 || del_pos > count) {
    printf("位置无效\n");
    return;
}

// 删除指定位置的节点
p = *head;
q = NULL;
index = 1;
while (p != NULL && index != positions[del_pos - 1]) {
    q = p;
    p = p->next;
    index++;
}

if (q == NULL) {
    *head = p->next;
} else {
    q->next = p->next;
}

// 更新后续节点的计数
for (struct node* temp = p->next; temp != NULL; temp = temp->next) {
    temp->count--;
}

free(p);

}

// 搜索节点并返回所有匹配节点的位置
void search(struct node* head, int key, int* positions, int* count)
{
count = 0;
int index = 1;
for (struct node p = head; p != NULL; p = p->next)
{
if (p->data == key)
{
positions[*count] = index;
(*count)++;
}
index++;
}
}

int main()
{
printf(“请先创建一个链表：\n”);
int n;
printf(“输入链表的长度: “);
scanf(“%d”, &n);

// 动态创建链表并赋值
struct node* head = NULL;
struct node* temp = NULL;
int value[n];
printf("依次输入%d个节点的数据: ", n);
for (int i = 0; i < n; i++) 
{
    scanf("%d", &value[i]);
}
    for (int i = 0; i < n; i++) 
{
    struct node* new_node = create_node(value[i], i + 1);
    if (head == NULL) 
    {
        head = new_node;
        temp = head;
    }
    else 
    {
        temp->next = new_node;
        temp = temp->next;
    }
}

printf("当前链表数据依次为：");
print_list(head);

while (1) 
{
    char choice[10];
    printf("请选择操作(插入/删除/搜索/打印/退出): ");
    scanf("%s", choice);
    getchar();

    if (strcmp(choice, "插入") == 0) 
    {
        printf("你想要插入几次:");
        int m;
        scanf("%d", &m);
        getchar();
        for (int i = 0; i < m; i++)
        {
            int pos, new_data;
            printf("输入要插入节点的位置(从1开始)和数据: ");
            scanf("%d %d", &pos, &new_data);
            getchar();
            if (pos < 1 || pos > n) 
            {
                printf("位置超出链表长度\n");
                continue;
            }
            temp = head;
            for (int i = 1; i < pos; i++) 
            {
                if (temp == NULL) 
                {
                    printf("位置超出链表长度\n");
                    break;
                }
                temp = temp->next;
            }
            insert_node(temp, new_data);
            printf("插入新节点后的链表数据为：");
            print_list(head);
            n++; // 更新链表长度
        } 
    }
    else if (strcmp(choice, "删除") == 0) 
    {
        printf("你想要删除几次:");
        int m;
        scanf("%d", &m);
        getchar();
        for (int i = 0; i < m; i++)
        {
            int value_to_delete;
            printf("输入要删除的节点的数据: ");
            scanf("%d", &value_to_delete);
            getchar();
            delete_node(&head, value_to_delete);
            printf("删除节点后的链表数据为：");
            print_list(head);
            n--; // 更新链表长度
        }
    } 
    else if (strcmp(choice, "搜索") == 0) 
    {
        printf("你想要搜索几次:");
        int m;
        scanf("%d", &m);
        getchar();
        for (int i = 0; i < m; i++)
        {
            int key;
            printf("输入要搜索的节点数据：");
            scanf("%d", &key);
            getchar();
            int positions[n];
            int count;
            search(head, key, positions, &count);
            if (count > 0) 
            {
                printf("找到%d个节点,位置分别为:", count);
                for (int i = 0; i < count; i++) 
                {
                    printf("%d ", positions[i]);
                }
                printf("\n");
            } 
            else 
            {
                printf("未找到节点\n");
            }
        }    
    } 
    else if (strcmp(choice, "退出") == 0) 
    {
        struct node* current = head;
        while (current != NULL) 
        {
            struct node* next = current->next;
            free(current);
            current = next;
        }
        printf("觉得我做的好就给我点个赞哦！！！\n");
        return 0;
    } 
    else if (strcmp(choice, "打印") == 0)
    {
        printf("当前链表数据依次为：");
        print_list(head);
    }
    else 
    {
        printf("没有该选项哦，请重新输入：\n");
    }
}
system("cls")
return 0;

}
